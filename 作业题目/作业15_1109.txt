    5.1-1 指针基本语法示例
    #include <stdio.h>
    int main()
    {
     float x=3.14159;
     float *p; //定义指针变量 p
     p = &x; //用 p 保存变量 x 的地址
     printf("%f\n", *p); //用 p 从地址中取内容
     return 0;
    }
